import com.koupper.octopus.process.ScriptProcessor

val setup: (ScriptProcessor) -> Unit = { processor ->
    processor.register(
        "process-project-name",
        mutableMapOf(
            "server.port" to "8080",
            "server.rootUrl" to "api/v1"
        ),
        "LOCAL_AWS_DEPLOYMENT",
        "1.0.0",
        "dev.example",
        mapOf(
            "idForTheScript" to "relative/path/to/script.kts",
        )
    ).run()
}
